CREATE TABLE Users (
    ID INT PRIMARY KEY IDENTITY,
    Username VARCHAR(100) NOT NULL,
    Password VARBINARY(256) NOT NULL,
    Email VARCHAR(100) UNIQUE NOT NULL,
    Permission INT NOT NULL,
    Sex INT NOT NULL,
    DateOfBirth DATE NOT NULL
);

CREATE TABLE Friends (
	ID1 INT,
	ID2 INT,
	PRIMARY KEY(ID1, ID2)
);

ALTER TABLE Friends
ADD CONSTRAINT FK_Friends_ID1 FOREIGN KEY (ID1) REFERENCES Users(ID);

ALTER TABLE Friends
ADD CONSTRAINT FK_Friends_ID2 FOREIGN KEY (ID2) REFERENCES Users(ID);

CREATE TABLE URLS (
	ID INT PRIMARY KEY,
	URL VARCHAR(100),
	FOREIGN KEY (ID) REFERENCES Users(ID)
);
